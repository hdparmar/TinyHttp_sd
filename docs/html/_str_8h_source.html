757 static char* itoa(int n, char s[]) {
758 int i, sign;
759 if ((sign = n) < 0) /* record sign */
760 n = -n; /* make n positive */
761 i = 0;
762 do { /* generate digits in reverse order */
763 s[i++] = n % 10 + '0'; /* get next digit */
764 } while ((n /= 10) > 0); /* delete it */
765 if (sign < 0)
766 s[i++] = '-';
767 s[i] = '\0';
768 reverse(s);
769 return s;
770 }
